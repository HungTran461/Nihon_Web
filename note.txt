function animateSVGSequential(svg) {
  const paths = Array.from(svg.querySelectorAll('path'));

  // Chuẩn bị tất cả path
  paths.forEach(path => {
    const length = path.getTotalLength();
    path.style.strokeDasharray = length;
    path.style.strokeDashoffset = length;
    path.style.opacity = 0;
    path.style.transition = 'none';
  });

  let index = 0;

  function drawNextPath() {
    if (index >= paths.length) return;

    const path = paths[index];
    const length = path.getTotalLength();

    // Bỏ qua path không có stroke hoặc length = 0
    const stroke = window.getComputedStyle(path).stroke;
    if (length === 0 || stroke === 'none') {
      index++;
      drawNextPath();
      return;
    }

    // Hiện path
    path.style.opacity = 1;

    // Force reflow
    path.getBoundingClientRect();

    // Bắt đầu animation
    const duration = 2000; // 2s
    path.style.transition = `stroke-dashoffset ${duration}ms ease-out`;
    path.style.strokeDashoffset = 0;

    // Dùng timeout thay vì transitionend
    setTimeout(() => {
      index++;
      drawNextPath();
    }, duration + 50);
  }

  drawNextPath();
}

// ✅ Gọi thử cho SVG đầu tiên trên trang
// Bạn có thể chạy dòng này sau khi paste code trên:
animateSVGSequential(document.querySelector('svg'));
